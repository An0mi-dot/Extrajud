    <script type="module">
        import { supabase } from './assets/supabaseClient.js';
        const { ipcRenderer } = require('electron');

        // Switch Account Logic
        window.switchAccount = async () => {
            try {
                // Find Supabase token key
                const sessionKey = Object.keys(localStorage).find(k => k.startsWith('sb-') && k.endsWith('-auth-token'));
                if(sessionKey) {
                    sessionStorage.setItem('backup_session_key', sessionKey);
                    sessionStorage.setItem('backup_session_val', localStorage.getItem(sessionKey));
                }
            } catch(e) { console.error('Backup session failed', e); }
            
            localStorage.clear();
            window.location.href = 'login.html?from=switch_account';
        };

        // --- User Global Context ---
        let userData = null;

        document.addEventListener('DOMContentLoaded', async () => {
             // Update Version Globally
             try {
                const ver = await ipcRenderer.invoke('get-app-version') || '2.0.6';
                document.querySelectorAll('.version-tag').forEach(el => el.textContent = 'ver ' + ver);
                document.querySelectorAll('.app-version, .settings-ver-text, #settings-app-ver').forEach(el => el.textContent = ver);
             } catch(e) { console.error('Version fetch failed', e); }

             // --- Update Notification ---
             const btnUpdateNotify = document.getElementById('btn-update-notify');
             const modalUpdate = document.getElementById('modal-update');
             const updateModalInfo = document.getElementById('update-modal-info');
             let updateUrl = null;

             function showUpdateBadge(newVer, url) {
                 updateUrl = url;
                 if (updateModalInfo) updateModalInfo.textContent = `Versão ${newVer} disponível. Clique para baixar.`;
                 if (btnUpdateNotify) { btnUpdateNotify.style.display = 'flex'; }
             }

             // Listen for electron-updater events
             ipcRenderer.on('updater:status', (_, data) => {
                 if (data.status === 'available' || data.status === 'ready') {
                     const newVer = (data.info && data.info.version) ? data.info.version : '';
                     showUpdateBadge(newVer, null);
                 }
             });

             // Manual check via updates.json
             try {
                 const upd = await ipcRenderer.invoke('check-for-updates');
                 if (upd && upd.updateAvailable) {
                     showUpdateBadge(upd.latestVersion, upd.url);
                 }
             } catch(e) { console.warn('Update check failed', e); }

             if (btnUpdateNotify) {
                 btnUpdateNotify.onclick = () => { if (modalUpdate) modalUpdate.style.display = 'flex'; };
             }
             document.getElementById('btn-update-dismiss')?.addEventListener('click', () => {
                 if (modalUpdate) modalUpdate.style.display = 'none';
             });
             document.getElementById('btn-update-download')?.addEventListener('click', () => {
                 const target = updateUrl || 'https://an0mi-dot.github.io/Extrajud/';
                 require('electron').shell.openExternal(target);
                 if (modalUpdate) modalUpdate.style.display = 'none';
             });

             // 1. Setup Header & Modal
            const header = document.getElementById('user-header-area');
            const nameEl = document.getElementById('display-name');
            const btnSaveProfile = document.getElementById('btn-save-profile');
            const modalProfile = document.getElementById('modal-profile');
            const btnCloseProfile = document.getElementById('btn-close-profile');
            const btnOpenSettings = document.getElementById('btn-open-settings');

            // NEW SETTINGS VARS
            const modalSettings = document.getElementById('modal-settings');
            const btnCloseSettings = document.getElementById('btn-close-settings');
            const btnSaveSettings = document.getElementById('btn-save-settings');
            const inpGlobalLog = document.getElementById('inp-global-log');
            const btnSelLogDir = document.getElementById('btn-sel-log-dir');
            
            // Checkboxes
            const chkMinimize = document.getElementById('chk-minimize-tray');
            const chkAlwaysTop = document.getElementById('chk-always-top');
            const chkSoundFinish = document.getElementById('chk-sound-finish');
            const chkNotificationFinish = document.getElementById('chk-notification-finish');
            
            if (btnCloseProfile) btnCloseProfile.onclick = () => {
                modalProfile.style.display = 'none';
                document.body.style.overflow = 'auto';
            };
            if (btnCloseSettings) btnCloseSettings.onclick = () => {
                modalSettings.style.display = 'none';
                document.body.style.overflow = 'auto';
            };
            
            // Close modais when clicking outside
            window.addEventListener('click', (e) => {
                if (e.target === modalSettings) {
                    modalSettings.style.display = 'none';
                    document.body.style.overflow = 'auto';
                }
                if (e.target === modalProfile) {
                    modalProfile.style.display = 'none';
                    document.body.style.overflow = 'auto';
                }
            });

            if (btnOpenSettings) {
                btnOpenSettings.onclick = async () => {
                   modalSettings.style.display = 'flex';
                   document.body.style.overflow = 'hidden';
                   
                   // Init Settings Tabs Logic
                   const tabs = modalSettings.querySelectorAll('.settings-tab');
                   const panes = modalSettings.querySelectorAll('.settings-tab-pane');

                   tabs.forEach(tab => {
                      tab.onclick = () => {
                          tabs.forEach(t => t.classList.remove('active'));
                          panes.forEach(p => p.classList.remove('active'));
                          
                          tab.classList.add('active');
                          const target = tab.getAttribute('data-tab'); // geral, logs, sobre
                          
                          // Convert data-tab to ID
                          const realTarget = document.getElementById('tab-' + target);
                          if(realTarget) realTarget.classList.add('active');
                      };
                   });
                   
                   // Init Theme Buttons Logic inside modal
                   modalSettings.querySelectorAll('.theme-option').forEach(btn => {
                       btn.onclick = () => {
                           const mode = btn.getAttribute('data-theme');
                           if(window.themeConfig) window.themeConfig.apply(mode);
                       };
                   });

                   // Load Current Settings State
                   try {
                       const state = await ipcRenderer.invoke('load-app-state');
                       if (state && state.globalLogDir && inpGlobalLog) {
                           inpGlobalLog.value = state.globalLogDir;
                       }

                       const genConfig = await ipcRenderer.invoke('get-general-config');
                       if (chkMinimize) chkMinimize.checked = genConfig.minimizeToTray || false;
                       if (chkAlwaysTop) chkAlwaysTop.checked = genConfig.alwaysOnTop || false;
                       if (chkSoundFinish) chkSoundFinish.checked = genConfig.soundFinish || false;
                       if (chkNotificationFinish) chkNotificationFinish.checked = genConfig.notificationFinish || false;

                       // Populate version info
                       const ver = await ipcRenderer.invoke('get-app-version') || '1.0.0';
                       const verEls = document.querySelectorAll('#settings-app-ver, .settings-ver-text'); 
                       verEls.forEach(el => el.textContent = ver);

                       // Env Info
                       if(document.getElementById('env-node')) document.getElementById('env-node').textContent = process.versions.node;
                       if(document.getElementById('env-electron')) document.getElementById('env-electron').textContent = process.versions.electron;

                       if(document.getElementById('env-plat')) document.getElementById('env-plat').textContent = process.platform + ' (' + process.arch + ')';
                   } catch(e) { console.error('Error loading settings', e); }
                };
            }

            if (btnSelLogDir) {
                btnSelLogDir.onclick = async () => {
                    const dir = await ipcRenderer.invoke('dialog:openDirectory');
                    if (dir && inpGlobalLog) {
                        inpGlobalLog.value = dir;
                    }
                };
            }

            if (btnSaveSettings) {
                btnSaveSettings.onclick = async () => {
                    try {
                        // Save Path
                        const state = await ipcRenderer.invoke('load-app-state') || {};
                        if(inpGlobalLog) state.globalLogDir = inpGlobalLog.value;
                        ipcRenderer.send('save-app-state', state);

                        await ipcRenderer.invoke('set-general-config', {
                            minimizeToTray: chkMinimize ? chkMinimize.checked : false,
                            alwaysOnTop: chkAlwaysTop ? chkAlwaysTop.checked : false,
                            soundFinish: chkSoundFinish ? chkSoundFinish.checked : false,
                            notificationFinish: chkNotificationFinish ? chkNotificationFinish.checked : false
                        });

                        modalSettings.style.display = 'none';
                    } catch(e) {
                        await ipcAlert('Erro ao salvar: ' + e.message);
                    }
                };
            }

            // 2. Load User
            try {
                // Try Supabase Session first to get latest token/data
                const sessionRes = await supabase.auth.getSession();
                if (sessionRes && sessionRes.data && sessionRes.data.session) {
                     userData = sessionRes.data.session.user;
                     // Enhance with Role
                     const { data: roleData } = await supabase.from('user_roles').select('role').eq('user_id', userData.id).single();
                     if (roleData) userData.role = roleData.role;
                     
                     // Push to main process cache
                     ipcRenderer.send('update-user-cache', userData);
                } else {
                    // Fallback to Main Process Cache if offline or session missing
                    userData = await ipcRenderer.invoke('get-current-user');
                }
                
                if (userData) {
                    initUserInterface(userData);
                }
            } catch (e) {
                console.error('User load error', e);
            }

            // 3. Save Profile Logic
            if (btnSaveProfile) {
                btnSaveProfile.onclick = async () => {
                    const newName = document.getElementById('inp-full-name').value;
                    const newAvatar = document.getElementById('inp-avatar-url').value;
                    
                    const originalText = btnSaveProfile.innerHTML;
                    btnSaveProfile.textContent = 'Salvando...';
                    
                    try {
                        const { data, error } = await supabase.auth.updateUser({
                            data: { full_name: newName, avatar_url: newAvatar }
                        });

                        if (error) throw error;
                        
                        // Update Local
                        userData = data.user;
                        
                        // Fetch Role from DB to ensure security context is complete
                        const { data: roleData } = await supabase.from('user_roles').select('role').eq('user_id', userData.id).single();
                        if (roleData) userData.role = roleData.role;

                        ipcRenderer.send('update-user-cache', userData);
                        initUserInterface(userData);
                        
                        modalProfile.style.display = 'none';
                        await ipcAlert('Perfil atualizado com sucesso!');
                        
                    } catch (e) {
                        await ipcAlert('Erro ao atualizar: ' + e.message);
                    } finally {
                        btnSaveProfile.innerHTML = originalText;
                    }
                };
            }
        });

        function getDisplayName(user) {
            const meta = user.user_metadata || {};
            if (meta.full_name) return meta.full_name;
            if (meta.first_name) return `${meta.first_name} ${meta.last_name || ''}`.trim();
            // Fallback: converte "luana.moreira" → "Luana Moreira"
            if (user.email) {
                const cap = s => s.charAt(0).toUpperCase() + s.slice(1);
                return user.email.split('@')[0].split(/[._-]/).map(cap).join(' ');
            }
            return 'Usuário';
        }

        function getAvatarUrl(user) {
             const meta = user.user_metadata || {};
             return meta.avatar_url || null;
        }

        function getInitials(name) {
            const parts = name.split(' ').filter(p => p.length > 0);
            if (parts.length === 0) return '?';
            if (parts.length === 1) return parts[0].substring(0, 2).toUpperCase();
            return (parts[0][0] + parts[parts.length - 1][0]).toUpperCase();
        }

        function updateAvatarUI(elId, name, url) {
            const el = document.getElementById(elId);
            if (!el) return;
            
            if (url) {
                el.innerHTML = `<img src="${url}" alt="Avatar">`;
                el.style.backgroundColor = 'transparent';
                el.style.border = 'none';
            } else {
                el.textContent = getInitials(name);
                el.style.backgroundColor = 'var(--primary-color)';
            }
        }

        function initUserInterface(user) {
            const displayName = getDisplayName(user);
            const avatarUrl = getAvatarUrl(user);

            window.switchTab = (tabName) => {
                document.querySelectorAll('.profile-section').forEach(el => el.classList.remove('active'));
    
                // Remove active class from all tabs first
               const allTabs = document.querySelectorAll('.profile-tab');
               allTabs.forEach(t => t.classList.remove('active'));
               
               // Activate Section
               const section = document.getElementById(tabName);
               if(section) section.classList.add('active');

               // Activate Button
               // Find the tab that calls this function with this Argument
               // Simple text match since we don't have IDs on tabs
               for(const t of allTabs) {
                   if(t.getAttribute('onclick').includes(tabName)) {
                       t.classList.add('active');
                       break;
                   }
               }
            };

            // 1. Setup Header
            const header = document.getElementById('user-header-area');
            const nameEl = document.getElementById('display-name');
            const btnChangeEmail = document.getElementById('btn-change-email');
            const btnResetPass = document.getElementById('btn-reset-password');

            if (header && nameEl) {
                nameEl.textContent = displayName;
                updateAvatarUI('display-avatar', displayName, avatarUrl);
                header.style.display = 'flex';
                
                // Open Profile Modal
                header.addEventListener('click', () => {
                   document.getElementById('modal-profile').style.display = 'flex';
                   document.getElementById('inp-full-name').value = getDisplayName(user);
                   document.getElementById('inp-avatar-url').value = getAvatarUrl(user) || '';
                   document.getElementById('inp-email').value = user.email || ''; 
                   updateAvatarUI('modal-avatar-preview', getDisplayName(user), getAvatarUrl(user));

                   // Update Central de Contas UI
                   const accName = document.getElementById('account-current-name');
                   const accEmail = document.getElementById('account-current-email');
                   const accAvatar = document.getElementById('account-current-avatar');

                   if(accName) accName.textContent = getDisplayName(user);
                   if(accEmail) accEmail.textContent = user.email;
                   if(accAvatar) updateAvatarUI('account-current-avatar', getDisplayName(user), getAvatarUrl(user));

                   // Reset tab
                   window.switchTab('tab-general');
                });
            }

            // --- Avatar Upload Logic ---
            const inpFile = document.getElementById('inp-avatar-file');
            const loadingText = document.getElementById('avatar-loading');
            
            if(inpFile) {
                inpFile.onchange = async (e) => {
                    const file = e.target.files[0];
                    if(!file) return;
                    
                    if(file.size > 2 * 1024 * 1024) {
                        await ipcAlert('A imagem deve ter no máximo 2MB.');
                        return;
                    }
                    
                    if(loadingText) loadingText.style.display = 'block';
                    
                    try {
                        const fileExt = file.name.split('.').pop();
                        const fileName = `${user.id}-${Math.random()}.${fileExt}`;
                        const filePath = `${fileName}`;
                        
                        // 1. Upload to Supabase Storage 'avatars' bucket
                        const { error: uploadError } = await supabase.storage
                            .from('avatars')
                            .upload(filePath, file);

                        if (uploadError) throw uploadError;

                        // 2. Get Public URL
                        const { data } = supabase.storage.from('avatars').getPublicUrl(filePath);
                        const publicUrl = data.publicUrl;

                        // 3. Update User Metadata
                        // We also set the input value so the user sees it immediately
                        document.getElementById('inp-avatar-url').value = publicUrl;
                        updateAvatarUI('modal-avatar-preview', getDisplayName(user), publicUrl);
                        
                        await ipcAlert('Imagem enviada! Clique em SALVAR ALTERAÇÕES para confirmar.');

                    } catch (error) {
                        console.error('Upload error', error);
                        await ipcAlert('Erro ao enviar imagem: ' + (error.message || error));
                    } finally {
                        if(loadingText) loadingText.style.display = 'none';
                        inpFile.value = ''; // Reset input to allow sending same file again
                    }
                };
            }

            // --- Security Handlers ---

            if (btnChangeEmail) {
                btnChangeEmail.onclick = async () => {
                    const newEmail = prompt("Digite o novo email:");
                    if (!newEmail || !newEmail.includes('@')) return;
                    
                    const originalText = btnChangeEmail.textContent;
                    btnChangeEmail.textContent = '...';
                    btnChangeEmail.disabled = true;

                    try {
                        const { error } = await supabase.auth.updateUser({ email: newEmail });
                        if (error) throw error;
                        await ipcAlert(`Um email de confirmação foi enviado para ${newEmail} e para o seu email atual.`);
                    } catch (e) {
                        await ipcAlert('Erro ao alterar email: ' + e.message);
                    } finally {
                        btnChangeEmail.textContent = originalText;
                        btnChangeEmail.disabled = false;
                    }
                };
            }

            if (btnResetPass) {
                btnResetPass.onclick = async () => {
                    if (!await ipcConfirm("Isso enviará um link de redefinição para " + user.email + ". Continuar?")) return;
                    
                    const originalText = btnResetPass.innerHTML;
                    btnResetPass.textContent = 'Enviando...';
                    btnResetPass.disabled = true;

                    try {
                        const { error } = await supabase.auth.resetPasswordForEmail(user.email);
                        if (error) throw error;
                        await ipcAlert('Email enviado! Verifique sua caixa de entrada.');
                    } catch (e) {
                        await ipcAlert('Erro: ' + e.message);
                    } finally {
                        btnResetPass.innerHTML = originalText;
                        btnResetPass.disabled = false;
                    }
                };
            }


            // --- Debug Logic ---
            let isTestMode = false;
            let f5Timer = null;
            let f5Flag = false;

            const toggleTestMode = () => {
                isTestMode = !isTestMode;
                const fab = document.getElementById('debug-fab');
                if (fab) {
                    fab.style.display = isTestMode ? 'flex' : 'none';
                    // Reset menu when turning off
                    if (!isTestMode) {
                        const menu = document.getElementById('debug-menu');
                        if (menu) menu.style.display = 'none';
                    } else {
                        // Announce activation
                        ipcAlert('Modo de Teste: ' + (isTestMode ? 'ATIVADO' : 'DESATIVADO')).catch(()=>{});
                    }
                }
            };

            window.addEventListener('keydown', (e) => {
                if (e.key === 'F5') {
                    // Prevent default only if we are planning to use the combination? 
                    // Be careful not to block reload if user just wants to reload.
                    // But for now, user requested F5+T
                    // Let's not prevent Default immediately unless needed, or just warn user
                    // e.preventDefault(); 
                    f5Flag = true;
                    // We don't prevent default here to allow normal F5 refresh if T is not pressed quickly
                    // But if T is pressed, we might be too late. 
                    // Actually, if F5 is held down, it will refresh repeatedly. 
                    // In Electron PROD, F5 is usually disabled. In DEV, it's useful.
                    // Let's assumes user holds F5.
                    clearTimeout(f5Timer);
                    f5Timer = setTimeout(() => { f5Flag = false; }, 500);
                }
                if ((e.key === 't' || e.key === 'T') && f5Flag) {
                    e.preventDefault(); // Prevent 't' being typed if focused somewhere
                    f5Flag = false;
                    toggleTestMode();
                }
            });

            const fab = document.getElementById('debug-fab');
            const menu = document.getElementById('debug-menu');
            const btnForce = document.getElementById('btn-force-welcome');

            if(fab) {
                fab.addEventListener('click', () => {
                   if(menu) menu.style.display = (menu.style.display === 'block') ? 'none' : 'block';
                });
            }

            if(btnForce) {
                btnForce.addEventListener('click', () => {
                    playWelcomeAnimation(displayName);
                    if(menu) menu.style.display = 'none';
                });
            }

            // 2. Welcome Screen Logic - refactored
            // Helper function
            window.playWelcomeAnimation = (name) => {
                const welcome = document.getElementById('welcome-overlay');
                const welcomeName = document.getElementById('welcome-name');
                const particles = document.getElementById('particles-area');

                if (welcome) {
                    const firstName = name.split(' ')[0];
                    if(welcomeName) welcomeName.textContent = firstName;
                    
                    // Reset styles for replay
                    welcome.style.display = 'flex';
                    welcome.style.opacity = '1';
                    welcome.style.filter = 'none';
                    
                    // Clear existing particles
                    if(particles) particles.innerHTML = '';

                    // Particle Generation
                    if(particles) {
                        for(let i=0; i<50; i++) {
                            const p = document.createElement('div');
                            p.className = 'particle';
                            p.style.left = Math.random() * 100 + '%';
                            p.style.top = Math.random() * 100 + '%'; // Start anywhere
                            // Randomize animation so they don't all start at once
                            p.style.animationDelay = (Math.random() * 2) + 's';
                            p.style.animationDuration = (Math.random() * 3 + 3) + 's'; // 3-6s
                            const size = Math.random() * 6 + 2;
                            p.style.width = size + 'px';
                            p.style.height = size + 'px';
                            p.style.opacity = Math.random() * 0.4 + 0.1;
                            particles.appendChild(p);
                        }
                    }

                    const closeWelcome = () => {
                         welcome.style.opacity = '0';
                         welcome.style.filter = 'blur(10px) brightness(1.2)'; // Smooth cinematic exit
                         // Wait for transition end
                         setTimeout(() => {
                             welcome.style.display = 'none';
                             welcome.style.filter = 'none'; // reset
                         }, 800);
                         sessionStorage.setItem('welcome_seen', 'true');
                    };

                    // Auto hide after 4.5s
                    // Clear any existing timeout if we are forcing it multiple times
                    if(window.welcomeTimeout) clearTimeout(window.welcomeTimeout);
                    
                    window.welcomeTimeout = setTimeout(() => {
                        if(welcome.style.display !== 'none') closeWelcome();
                    }, 4500);
                }
            };

            // Use sessionStorage to only show once per app launch session
            if (!sessionStorage.getItem('welcome_seen')) {
                playWelcomeAnimation(displayName);
            }
        }
    </script>
